### Scoring Logic (Pseudocode)

```javascript
// Calculate total score for a user
function calculateQualificationScore(userId) {
  const user = getUserData(userId);
  
  // 1. FARM SIZE SCORE (0-40 points)
  let farmSizeScore = 0;
  if (user.farm_size >= 100000) farmSizeScore = 40;
  else if (user.farm_size >= 80000) farmSizeScore = 30;
  else if (user.farm_size >= 60000) farmSizeScore = 20;
  else if (user.farm_size >= 40000) farmSizeScore = 10;
  
  // 2. GROWTH RATE SCORE (0-20 points)
  let growthScore = 0;
  const growthRate = user.growth_rate; // % YoY
  if (growthRate >= 50) growthScore = 20;
  else if (growthRate >= 30) growthScore = 15;
  else if (growthRate >= 15) growthScore = 10;
  else if (growthRate >= 5) growthScore = 5;
  
  // 3. ENGAGEMENT SCORE (0-20 points)
  const toolsUsed = countToolsUsedRegularly(userId); // last 30 days
  let engagementScore = 0;
  if (toolsUsed >= 7) engagementScore = 20;
  else if (toolsUsed >= 5) engagementScore = 15;
  else if (toolsUsed >= 3) engagementScore = 10;
  else if (toolsUsed >= 1) engagementScore = 5;
  
  // 4. PAIN SIGNAL SCORE (0-20 points)
  const signals = getRecentSignals(userId, 90); // last 90 days
  let painSignalScore = 0;
  
  signals.forEach(signal => {
    switch(signal.type) {
      case 'explicit_upgrade_interest':
        painSignalScore += 20;
        break;
      case 'batch_limitation_hit':
        painSignalScore += 15;
        break;
      case 'viewed_upgrade_page':
        painSignalScore += 10;
        break;
      case 'watched_demo_video':
        painSignalScore += 15;
        break;
      case 'downloaded_comparison':
        painSignalScore += 10;
        break;
      // ... more signal types
    }
  });
  
  // Cap pain signals at 20
  painSignalScore = Math.min(painSignalScore, 20);
  
  // 5. TOTAL SCORE
  const totalScore = farmSizeScore + growthScore + engagementScore + painSignalScore;
  
  // 6. DETERMINE SEGMENT
  let segment;
  if (totalScore >= 90) segment = 'hot';
  else if (totalScore >= 70) segment = 'warm';
  else if (totalScore >= 50) segment = 'potential';
  else segment = 'too_early';
```
